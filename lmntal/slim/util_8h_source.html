123 inline unsigned long hash(const unsigned char *str, long i) {
124  /*
125  * FNV-1a hash each octet in the buffer
126  */
127  auto hval = fnv::basis();
128  while (--i >= 0) {
129  /* xor the bottom with the current octet */
130  hval ^= (unsigned int)str[i];
131  /* multiply by the FNV magic prime mod 2^32 or 2^64 */
132  hval *= fnv::prime();
133  }
134 
135  return hval;
136 }

166 /* n以上で最小の2の倍数を返す */
167 static inline unsigned long round2up(unsigned long n) {
168  unsigned int v = 1;
169  while (v && v < n) {
170  v <<= 1;

186 public:

194  raw_pointer_iterator(const raw_pointer_iterator<T> &it) : p(it.p) {}

203  p++;

207  auto it = *this;

210  }
211 
212  bool operator!=(const raw_pointer_iterator<T> &a) { return !(*this == a); }
213  bool operator==(const raw_pointer_iterator<T> &a) { return p == a.p; }
