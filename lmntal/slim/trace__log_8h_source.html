66 class LogTracker {
67  std::stack<ProcessID> traced_ids;
68  std::stack<std::size_t> btp_idx;
69 
70 public:
71  void push() { btp_idx.push(traced_ids.size()); }
72 
73  void pop() { btp_idx.pop(); }
74 
75  void trace(ProcessID id) { traced_ids.push(id); }
76 
77  /* 前回のトレースプロセス数を基に, ログを巻き戻す. */
78  template <typename T> void revert(T table) {
79  /* 前回セットしたトレース数のメモを取り出す */
80  const auto bt_idx = btp_idx.top();
81  btp_idx.pop();
82  /* bt_idxより上のidxに積まれたデータをログ上からunputしていく */
83  while (traced_ids.size() > bt_idx) {
84  auto key = traced_ids.top();
85  traced_ids.pop();
86  table->leave(key);
87  }
88  }
89 };

96 /* 各{シンボルアトム, inside proxyアトム, 膜}に対して1つずつ対応させるデータ構造
97  * outside proxyアトムは含めない */
98 struct TraceData { /* 64bit: 24Bytes (32bit: 16Bytes) */
99  BYTE flag; /* 対応させているデータの種類を示すフラグ */
100  unsigned int traversed_proc; /* 膜に対応させている場合は, その膜内で訪問した
101  {シンボルアトム, inside proxyアトム,
102  子膜}の総数 を記録している.
103  他のデータに対応させている場合は0のまま */
104  ProcessID owner_id; /* 対応させているデータ構造の所属膜のID.
105  * プロセスIDは1から始まるため,
106  * 所属膜がない(例えばグローバルルート膜の)場合は, 0 */
107  ProcessID matched; /* 対応させているプロセスとマッチさせたプロセスのID.
108  * in-proxyアトムはBS encode時の訪問順序に数えないため,
109  * in-proxyアトムへの対応としては0をセット */
110 
111  TraceData() : flag(0), traversed_proc(0), owner_id(0), matched(0) {}
112  TraceData(BYTE flag, unsigned int traversed_proc, ProcessID owner_id,
113  ProcessID matched)
114  : flag(flag), traversed_proc(traversed_proc), owner_id(owner_id),
115  matched(matched) {}
116 
117  bool operator==(const TraceData &a) const {
118  return a.flag == flag && a.traversed_proc == traversed_proc &&
119  a.owner_id == owner_id && a.matched == matched;
120  }
121  bool operator!=(const TraceData &a) const { return !(*this == a); }
122 
123  enum options {
124  TRAVERSED_ATOM = 0x1U,
125  TRAVERSED_MEM = 0x2U,
126  TRAVERSED_HLINK = 0x3U,
127  TRAVERSED_OTHERS = 0xfU,
128  };
129 };

131 struct TraceLog {
132  std::unordered_map<ProcessID, TraceData> table;
133  LogTracker tracker;
134 
135  using key_type = ProcessTable<TraceData>::key_type;
136 
137  TraceLog(unsigned long size) {}
138  TraceLog() {}
139  unsigned int traversed_proc_count(LmnMembraneRef owner) {
140  return table.find(owner->mem_id()) != table.end()
141  ? table[owner->mem_id()].traversed_proc
142  : 0;
143  }
144 
145  /* 膜ownerを対象として訪問済みにしたプロセス (シンボルアトム + 子膜 + inside
146  * proxies) の数が 膜ownerのそれと一致しているか否かを返す */
147  bool eq_traversed_proc_num(LmnMembraneRef owner, AtomListEntryRef in_ent,
148  AtomListEntryRef avoid) {
149  size_t s1 = in_ent ? in_ent->size() : 0;
150  size_t s2 = avoid ? avoid->size() : 0;
151  return traversed_proc_count(owner) ==
152  (owner->symb_atom_num() + owner->child_mem_num() + s1 - s2);
153  }
154 
155 private:
156  /* ログl上のキーkey位置にあるTraceLogに対して, 訪問を記録する.
157  * matched_idはマッチしたプロセスのID, ownerは所属膜.
158  * 所属膜がNULLでない場合は, 所属膜の情報を記録し,
159  * 所属膜側のプロセス訪問カウンタを回す.
160  */
161  bool visit(LmnWord key, BYTE flag, LmnWord matched_id, LmnMembraneRef owner) {
162  if (table.find(key) != table.end()) {
163  return false;
164  }
165 
166  TraceData value = {flag, 0, 0, matched_id};
167 
168  if (owner) {
169  value.owner_id = owner->mem_id();
170  table[owner->mem_id()].traversed_proc++;
171  }
172 
173  table.insert(std::make_pair(key, value));
174 
175  tracker.trace(key);
176 
177  return true;
178  }
179 
180 public:
181  /* ログに, 所属膜ownerのアトムatomへの訪問を記録する.
182  * atomにマッチしたアトムのプロセスIDもしくは訪問番号atom2_idを併せて記録する.
183  */
184  bool visit_atom(LmnSymbolAtomRef atom, LmnWord atom2_id,
185  LmnMembraneRef owner) {
186  return this->visit(atom->get_id(), TraceData::options::TRAVERSED_ATOM,
187  atom2_id, owner);
188  }
189 
190  /* ログに, 膜mem1への訪問を記録する.
191  * (所属膜はmem1のメンバから参照するため不要)
192  * mem1にマッチした膜のプロセスIDもしくは訪問番号mem2_idを併せて記録する */
193  bool visit_mem(LmnMembraneRef mem1, LmnWord mem2_id) {
194  return this->visit(mem1->mem_id(), TraceData::options::TRAVERSED_MEM,
195  mem2_id, mem1->mem_parent());
196  }
197 
198  /* ログに, ハイパーグラフのルートオブジェクトhl1への訪問を記録する.
199  * (ハイパーグラフ構造には所属膜の概念がなく,
200  * 膜オブジェクトからの参照もできないため, 所属膜に対する一切の操作は不要)
201  * hl1にマッチしたハイパリンクオブジェクトIDもしくは訪問番号hl2_idを併せて記録する
202  */
203  bool visit_hlink(HyperLink *hl1, LmnWord hl2_id) {
204  return this->visit(LMN_HL_ID(hl1), TraceData::options::TRAVERSED_HLINK,
205  hl2_id, NULL);
206  }
207 
208  void leave(key_type key) {
209  const auto owner_id = table[key].owner_id;
210  table[owner_id].traversed_proc--;
211  if (table[owner_id] == TraceData())
212  table.erase(owner_id);
213  table.erase(key);
214  }
215 
216  bool contains(key_type key) const {
217  return table.find(key) != table.end();
218  }
219 
220  void backtrack() { tracker.revert(this); }
221 
222  void set_btpoint() { tracker.push(); }
223 
224  void continue_trace() { tracker.pop(); }
225 };
