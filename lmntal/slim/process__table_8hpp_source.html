52 /* LMNtalのプロセス（アトム、膜）をキーにもちいるテーブル */
53 template <typename T> class ProcessTable {
54 public:
55  using key_type = ProcessID;
56  using value_type = T;
57 
58  const static key_type not_found;
59  static constexpr std::size_t buckets_size = 1 << 12; // heuristics
60 
61 private:
62  const static value_type unused;
63  unsigned long n;
64  unsigned long size;
65  unsigned long num_buckets;
66  value_type **tbl;
67 
68 public:
69  ProcessTable(unsigned long size = 64) {
70  this->n = 0;
71  this->size = size ? size : 1;
72  this->num_buckets = size / buckets_size + 1;
73  this->tbl = LMN_CALLOC(value_type *, this->num_buckets);
74  }
75 
76  ~ProcessTable() {
77  for (int i = 0; i < this->num_buckets; i++) {
78  LMN_FREE(this->tbl[i]);
79  }
80  LMN_FREE(this->tbl);
81  }
82 
83  bool operator==(const ProcessTable<value_type> *b) {
84  if (this->n != b->n)
85  return false;
86 
87  unsigned int a_checked = 0;
88 
89  for (int i = 0; i < this->num_buckets; i++) {
90  if (!this->tbl[i] && !b->tbl[i])
91  continue;
92 
93  for (int j = 0; j < buckets_size && a_checked < this->n; j++) {
94  value_type va = (this->tbl[i]) ? this->tbl[i][j] : unused;
95  value_type vb = (b->tbl[i]) ? b->tbl[i][j] : unused;
96  if (va != vb)
97  return false;
98  if (va != unused)
99  a_checked++;
100  }
101  }
102 
103  return true;
104  }
105 
106  value_type &operator[](key_type key) {
107  expand(key);
108  return this->tbl[key / buckets_size][key % buckets_size];
109  }
110 
111  template <typename U> bool contains(U key) const {
112  const auto k = slim::process_id(key);
113  return k < this->size && this->tbl[k / buckets_size] &&
114  this->tbl[k / buckets_size][k % buckets_size] != unused;
115  }
116 
117  template <typename U> void put(U key_, value_type value) {
118  const auto key = slim::process_id(key_);
119 #ifdef DEBUG
120  if (value == unused)
121  lmn_fatal("cannot put 'unused' value.");
122 #endif
123  if (!this->contains(key)) {
124  this->n++;
125  this->expand(key);
126  }
127  this->tbl[key / buckets_size][key % buckets_size] = value;
128  }
129 
130  /* テーブルにkeyを追加し, trueを返す. すでにpが存在した場合はfalseを返す. */
131  template <typename U> bool put_if_absent(U key, value_type value) {
132  if (this->contains(key))
133  return false;
134  this->put(key, value);
135  return true;
136  }
137 
138  template <typename U> void unput(U key) {
139  const auto k = slim::process_id(key);
140 #ifdef DEBUG
141  if (!this->contains(k))
142  throw new std::logic_error("attempted to unput an absent key.");
143 #endif
144  this->n--;
145  this->tbl[k / buckets_size][k % buckets_size] = unused;
146  }
147 
148  template <typename U> void erase(U key) { unput(key); }
149 
150  void clear() {
151  this->n = 0;
152  for (int i = 0; i < this->num_buckets; i++) {
153  if (this->tbl[i])
154  memset(this->tbl[i], 0xff, sizeof(value_type) * buckets_size);
155  }
156  }
157 
158  void foreach (int (*func)(key_type key, value_type val, LmnWord arg),
159  LmnWord arg) {
160  unsigned long n = 0;
161 
162  for (int i = 0; i < this->num_buckets; i++) {
163  if (!this->tbl[i])
164  continue;
165  for (int j = 0; j < buckets_size && n < this->n; j++) {
166  if (this->tbl[i][j] == unused)
167  continue;
168  func(i * buckets_size + j, this->tbl[i][j], arg);
169  n++;
170  }
171  }
172  }

174  class iterator : std::input_iterator_tag {
175  const ProcessTable<value_type> *table;
176  std::size_t bucket_idx, idx;
177  std::pair<key_type, value_type> value;
178 
179  public:
180  iterator(const ProcessTable<value_type> *table) : table(table), bucket_idx(not_found), idx(not_found) {}
181  iterator(const ProcessTable<value_type> *table, std::size_t bucket_idx,
182  std::size_t idx)
183  : table(table), bucket_idx(bucket_idx), idx(idx),
184  value(std::make_pair(bucket_idx * buckets_size + idx,
185  table->tbl[bucket_idx][idx])){};
186 
187  const std::pair<key_type, value_type> &operator*() const { return value; }
188 
189  const std::pair<key_type, value_type> *operator->() const { return &value; }
190 
191  iterator &operator++() {
192  for (; bucket_idx < table->num_buckets; bucket_idx++) {
193  if (!table->tbl[bucket_idx])
194  continue;
195  for (idx++; idx < buckets_size; idx++) {
196  if (table->tbl[bucket_idx][idx] == unused)
197  continue;
198  value = std::make_pair(bucket_idx * buckets_size + idx,
199  table->tbl[bucket_idx][idx]);
200  return *this;
201  }
202  }
203  bucket_idx = not_found;
204  idx = not_found;
205  return *this;
206  }
207 
208  iterator operator++(int _) {
209  iterator it(this->table, bucket_idx, idx);
210  ++(*this);
211  return it;
212  }
213 
214  bool operator==(const iterator &it) const {
215  return table == it.table && bucket_idx == it.bucket_idx && idx == it.idx;
216  }
217  bool operator!=(const iterator &it) const { return !(*this == it); }
218  };
219  friend iterator;
220 
221  iterator begin() {
222  for (int i = 0; i < this->num_buckets; i++) {
223  if (!this->tbl[i])
224  continue;
225  for (int j = 0; j < buckets_size; j++) {
226  if (this->tbl[i][j] == unused)
227  continue;
228  return iterator(this, i, j);
229  }
230  }
231  return end();
232  }
233 
234  iterator end() const { return iterator(this); }
235 
236  template <typename U> iterator find(U key) const {
237  const auto k = slim::process_id(key);
238  return (k < this->size && this->tbl[k / buckets_size] &&
239  this->tbl[k / buckets_size][k % buckets_size] != unused)
240  ? iterator(this, k / buckets_size, k % buckets_size)
241  : end();
242  }
243 
244 private:
245  void expand(unsigned long n) {
246  unsigned int org_n = this->num_buckets;
247  while (this->size <= n)
248  this->size *= 2;
249  this->num_buckets = this->size / buckets_size + 1;
250  if (org_n < this->num_buckets) {
251  this->tbl = LMN_REALLOC(value_type *, this->tbl, this->num_buckets);
252  memset(this->tbl + org_n, 0,
253  sizeof(value_type *) * (this->num_buckets - org_n));
254  }
255 
256  unsigned int b = n / buckets_size;
257  if (b < this->num_buckets && this->tbl[b])
258  return;
259  this->tbl[b] = LMN_NALLOC(value_type, buckets_size);
260  for (int i = 0; i < buckets_size; i++)
261  this->tbl[b][i] = unused;
262  }
263 };
