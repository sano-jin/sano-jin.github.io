211  に、その膜がスタックに積まれている事がある。そのため、安全な削除を行

213  はこの手続きは必要ない*/
214  void memstack_remove(LmnMembrane *mem) {
215  memstack.erase(std::remove(memstack.begin(), memstack.end(), mem), memstack.end());
216  }

218  memstack.clear();
219  memstack_reconstruct_rec(mem);
220  }

222 private:
223  /* 親膜を子膜よりも先に積む */
224  void memstack_reconstruct_rec(LmnMembrane *parent) {
225  memstack_push(parent);
226  for (auto m = parent->mem_child_head(); m; m = m->mem_next()) {
227  memstack_reconstruct_rec(m);
