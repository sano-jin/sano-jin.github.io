61  /* write_atomsの膜バージョン.
62  * ここで書き込む計算する分子には膜のみが含まれている */
63  void write_mems(LmnMembraneRef mem, BinStrCursor &bsp, VisitLogRef visited) {
64  BinStrCursor last_valid_bsp;
65  CheckpointRef last_valid_checkpoint = nullptr;
66 
67  if (!bsp.is_valid())
68  return;
69 
70  bool last_valid = false;
71  for (auto m = mem->mem_child_head(); m; m = m->mem_next()) {
72  if (visited->get_mem(m, NULL))
73  continue;
74 
75  BinStrCursor new_bsptr = bsp;
76 
77  visited->set_checkpoint();
78 
79  write_mem(m, 0, -1, -1, new_bsptr, visited, TRUE);
80 
81  if (new_bsptr.is_valid()) {
82  /* mからたどった分子が書き込みに成功したので、last_validに記憶する */
83  if (last_valid) {
84  delete last_valid_checkpoint;
85  }
86  last_valid_bsp = new_bsptr;
87  last_valid_checkpoint = visited->pop_checkpoint();
88  last_valid = true;
89  } else {
90  visited->revert_checkpoint();
91  }
92  }
93 
94  if (last_valid) {
95  /* 書き込みに成功した分子をログに記録して、次の分子に進む */
96  visited->push_checkpoint(last_valid_checkpoint);
97  write_mems(mem, last_valid_bsp, visited);
98 
99  if (last_valid_bsp.is_valid()) {
100  bsp = last_valid_bsp;
101  visited->commit_checkpoint();
102  } else {
103  visited->revert_checkpoint();
104  }
105  }
106  }

108  /* 膜memの全てのアトムのバイナリストリングを書き込む.
109  * 辿ってきたアトムfrom_atomとそのアトムの属性attrとこちらからのリンク番号fromを受け取る.
110  */
111  void write_mem(LmnMembraneRef mem, LmnAtomRef from_atom, LmnLinkAttr attr,
112  int from, BinStrCursor &bsp, VisitLogRef visited, BOOL is_id) {
113  LmnWord n_visited;
114 
115  if (!bsp.is_valid())
116  return;
117 
118  /* 訪問済み */
119  if (visited->get_mem(mem, &n_visited)) {
120  bsp.push_visited_mem(n_visited);
121 
122  if (from_atom) { /* 引き続きアトムをたどる */
123  write_mol(from_atom, attr, from, bsp, visited, is_id);
124  }
125  return;
126  }
127 
128  visited->put_mem(mem);
129  bsp.push_start_mem(mem->NAME_ID());
130 
131  if (!bsp.is_valid())
132  return;
133 
134  if (from_atom) {
135  write_mol(from_atom, attr, from, bsp, visited, is_id);
136  }
137 
138  /* アトム・膜・ルールセットの順に書込み */
139  if (is_id) { /* 膜に対して一意なIDとなるバイナリストリングへエンコードする場合
140  */
141  write_mem_atoms(mem, bsp, visited);
142  write_mems(mem, bsp, visited);
143  } else { /* 単なるバイナリストリングへエンコードする場合 */
144  dump_mem_atoms(mem, bsp, visited);
145  dump_mems(mem, bsp, visited);
146 
147  /* 膜memに存在するデータアトムを起点にしたinside
148  * proxyアトムをちゃんと書き込んでおく */
149 
150  auto ent = mem->get_atomlist(LMN_IN_PROXY_FUNCTOR);
151  if (ent) {
152  for (auto in : *ent) {
153  if (!LMN_ATTR_IS_DATA(in->get_attr(1)) ||
154  visited->get_atom(in, NULL)) {
155  continue;
156  }
157  /* -------------------------+
158  * [DATA ATOM]-0--1-[in]-0--|--0-[out]-1--..
159  * -------------------------+
160  */
161  bsp.push_escape_mem_data(in->get_link(1),
162  in->get_attr(1), visited);
163  auto out = (LmnSymbolAtomRef)in->get_link(0);
164  write_mol(out->get_link(1), out->get_attr(1),
165  LMN_ATTR_GET_VALUE(out->get_attr(1)), bsp,
166  visited, is_id);
167  };
168  }
169  }
170 
171  write_rulesets(mem, bsp);
172  bsp.push_end_mem();
173  }

175  /* アトムatomをバイナリストリングへ書き込む
176  * 入力:
177  * アトムatomと, atomのリンク属性attr,
178  * アトムatomへ辿ってきた際に辿ったリンクと接続するリンク番号from,
179  * エンコード領域bsp, 訪問管理visited,
180  * is_idは計算するバイナリストリングがmem_idならば真 */
181  void write_mol(LmnAtomRef atom, LmnLinkAttr attr, int from, BinStrCursor &bsp,
182  VisitLogRef visited, BOOL is_id) {
183  LmnWord n_visited;
184 
185  if (!bsp.is_valid())
186  return;
187 
188  /* データアトムの場合 */
189  if (LMN_ATTR_IS_DATA(attr)) {
190  bsp.push_data_atom(atom, attr, visited);
191  return;
192  }
193 
194  auto satom = reinterpret_cast<LmnSymbolAtomRef>(atom);
195  auto f = satom->get_functor();
196  if (f == LMN_OUT_PROXY_FUNCTOR) {
197  /* outside proxyの場合, inside proxy側の膜をwrite_memで書き込む */
198  auto in = (LmnSymbolAtomRef)satom->get_link(0);
199  auto in_mem = LMN_PROXY_GET_MEM(in);
200  if (visited->get_atom(in, NULL)) {
201  visited->put_atom(in);
202  }
203  write_mem(in_mem, in->get_link(1), in->get_attr(1),
204  LMN_ATTR_GET_VALUE(in->get_attr(1)), bsp, visited,
205  is_id);
206  } else if (f == LMN_IN_PROXY_FUNCTOR) {
207  /* inside proxyの場合, 親膜へ抜ける旨を示すタグTAG_ESCAPE_MEMを書き込む.
208  * その後, outside proxyから分子のトレース(write_mol)を引き続き実行する */
209  auto out = (LmnSymbolAtomRef)satom->get_link(0);
210  bsp.push_escape_mem();
211 
212  if (visited->get_atom(satom, NULL)) {
213  visited->put_atom(satom);
214  }
215 
216  write_mol(out->get_link(1), out->get_attr(1),
217  LMN_ATTR_GET_VALUE(out->get_attr(1)), bsp, visited,
218  is_id);
219  } else if (!visited->get_atom(satom, &n_visited)) {
220  /* 未訪問のシンボルアトムの場合 */
221  visited->put_atom(satom);
222  bsp.push_atom(satom);
223  if (!bsp.is_valid())
224  return;
225 
226  auto arity = LMN_FUNCTOR_GET_LINK_NUM(f);
227  for (auto i_arg = 0; i_arg < arity; i_arg++) {
228  if (i_arg == from) { /* 辿ってきたリンクに接続 */
229  bsp.push_from();
230  continue;
231  }
232  write_mol(satom->get_link(i_arg),
233  satom->get_attr(i_arg),
234  LMN_ATTR_GET_VALUE(satom->get_attr(i_arg)), bsp,
235  visited, is_id);
236  }
237  } else {
238  /* 訪問済のシンボルアトムの場合 */
239  bsp.push_visited_atom(n_visited, from);
240  }
241  }

243  /* atomsに含まれるアトムを起点とする未訪問分子を、バイナリストリングが
244  最小となるように書き込む */
245  template <class Iterator>
246  void write_mols(Iterator begin, Iterator end, BinStrCursor &bsp,
247  VisitLogRef visited) {
248  if (!bsp.is_valid())
249  return;
250 
251  /* atoms中の未訪問のアトムを起点とする分子を、それぞれ試みる */
252  for (auto it = begin; it != end; ++it) {
253  LmnSymbolAtomRef atom = *it;
254 
255  if (LMN_IS_HL(atom))
256  continue;
257 
258  if (visited->get_atom(atom, NULL))
259  continue;
260 
261  BinStrCursor new_bsptr = bsp;
262  visited->set_checkpoint();
263 
264  write_mol(atom, LMN_ATTR_MAKE_LINK(0), -1, new_bsptr, visited, TRUE);
265  if (new_bsptr.is_valid()) {
266  /* atomからたどった分子が書き込みに成功したので次のアトムの書き込みを試みる */
267  write_mols(std::next(it), end, new_bsptr, visited);
268 
269  if (new_bsptr.is_valid()) {
270  bsp = new_bsptr;
271  visited->commit_checkpoint();
272  } else {
273  visited->revert_checkpoint();
274  }
275  break;
276  } else {
277  visited->revert_checkpoint();
278  }
279  }
280  }

282  void write_rulesets(LmnMembraneRef mem, BinStrCursor &bsp) {
283  /* ルールセットがルールセットIDでソートされていることに基づいたコード */
284  auto n = mem->ruleset_num();
285  if (n == 0)
286  return;
287 
288  bool has_uniq = FALSE;
289  /* TODO: uniqルールセットが存在するか否かを検査するためだけに
290  * O(ルールセット数)かかってしまうため.
291  * ルールの移動や複製を行うプログラムで非効率 */
292  for (auto i = 0; i < n; i++) {
293  if (lmn_mem_get_ruleset(mem, i)->has_unique()) {
294  has_uniq = TRUE;
295  break;
296  }
297  }
298 
299  if (!has_uniq) {
300  bsp.push_start_rulesets(n);
301 
302  for (auto i = 0; i < n; i++) {
303  bsp.push_ruleset(lmn_mem_get_ruleset(mem, i));
304  }
305  } else {
306  bsp.push_ruleset_uniq(mem, n);
307  }
308  }

317  /* アトム列atomsから, visitedに未登録のアトムに対し, write_molを行う
318  * つまり, mhash同様に, 各アトムを起点とした分子単位でエンコードを行っている
319  */
320  template <typename Container>
321  void dump_mols(const Container &atoms, BinStrCursor &bsp,
322  VisitLogRef visited) {
323  /* atoms中の未訪問のアトムを起点とする分子を、それぞれ試みる */
324  for (auto atom : atoms) {
325  if (visited->get_atom(atom, NULL) || LMN_IS_HL(atom))
326  continue;
327 
328  write_mol((LmnAtomRef)atom, LMN_ATTR_MAKE_LINK(0), -1, bsp, visited,
329  FALSE);
330  }
331  }
332 
333  /* 膜中心の計算単位. 未訪問の子膜Xに対して, 子膜の分子を書き込み,
334  * dump_memsへ再起する 兄弟膜は子膜内のプロセスを全て書き込んでから訪問される
335  */
336  void dump_mems(LmnMembraneRef mem, BinStrCursor &bsp,
337  VisitLogRef visited) {
338  for (auto m = mem->mem_child_head(); m; m = m->mem_next()) {
339  if (!visited->get_mem(m, NULL)) {
340  write_mem(m, 0, -1, -1, bsp, visited, FALSE);
341  }
342  }
343  }

346  static LmnBinStr *encode(LmnMembraneRef mem, unsigned int tbl_size = 0 /* hint */) {
347  encoder e(mem, false, tbl_size);
348  auto visited = &e.visit_log;
349  auto &bs = e.binstr;
350  auto &bsp = e.cur;
351 
352  e.write_mem_atoms(mem, *bsp, visited);
353  e.write_mems(mem, *bsp, visited);
354  e.write_rulesets(mem, *bsp);
355 
356  return e.binary_string();
357  }

359  static LmnBinStr *dump(LmnMembraneRef mem, unsigned long tbl_size = 0 /* hint */) {
360  encoder e(mem, true, tbl_size);
361  auto visitlog = &e.visit_log;
362  auto &bs = e.binstr;
363  auto &bsp = e.cur;
364 
365  e.dump_mem_atoms(mem, *bsp, visitlog); /* 1. アトムから */
366  e.dump_mems(mem, *bsp, visitlog); /* 2. 子膜から */
367  e.write_rulesets(mem, *bsp); /* 3. 最後にルール */
368 
369  return e.binary_string();
