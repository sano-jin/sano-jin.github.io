209  switch (attr) {

212  case LMN_DBL_ATTR:
213  return push(TAG_DBL_DATA) &&

215  case LMN_HL_ATTR:
216  return push_hlink(atom, log);
217  case LMN_SP_ATOM_ATTR:
218  if (sp_atom_encoder(atom)) {
219  auto bytes = sp_atom_encoder(atom)(atom);
220  auto type = LMN_SP_ATOM_TYPE(atom);
221  return push(TAG_SP_ATOM_DATA) && push(&type, sizeof(type) * TAG_IN_BYTE) && push(bytes);

297 #define BS_SET(a, pos, v) \
298  (((pos)&1) \
299  ? ((a)[(pos) >> 1] = ((a)[(pos) >> 1] & 0x0f) | ((v) << TAG_BIT_SIZE)) \
300  : ((a)[(pos) >> 1] = (v & 0x0f) | ((a)[(pos) >> 1] & 0xf0)))

302  (((pos)&1) ? ((a)[(pos) >> 1] & 0xf0) >> TAG_BIT_SIZE \

305 /* エンコード処理(計算中)に用いるバイナリストリング(作業領域) */
306 class BinStr {
307  BYTE *v; /* バイト列(128個で初期化) */
308  int size; /* バッファのサイズ（4ビット単位）: 現在のバイト列の大きさ(128 *
309  TAG_IN_BYTEで初期化) */
310  int cur; /* 書き込み位置（4ビット単位） : 次に書き込む位置(0で初期化) */
311 
312 public:
313  BinStr() {
314  size = BS_TBL_SIZE * TAG_IN_BYTE;
315  v = LMN_NALLOC(BYTE, size / TAG_IN_BYTE);
316  memset(v, 0x0U, sizeof(BYTE) * BS_TBL_SIZE);
317  cur = 0;
318  }
319 
320  ~BinStr() { LMN_FREE(v); }

330 public:
331  /* bsの位置posにbの下位4ビットを書き込む。書き込みに成功した場合は真を
332  返し、失敗した場合は偽を返す。*/
333  bool write(BYTE b, int pos) {
334  resize(pos);
335 
336  /* cur==posならば無条件で書き込み成功 */
337  if (cur == pos) {
338  BS_SET(v, pos, b);
339  cur++;
340  return true;
341  } else if (cur > pos) {
342  /* 下位4ビットの比較を行う */
343  if (BS_GET(v, pos) < b)
344  return false;
345  else if (BS_GET(v, pos) > b) {
346  cur = pos + 1;
347  BS_SET(v, pos, b);
348  return true;
349  } else { /* v[pos] == b */
350  return true;
351  }
352  } else { /*(cur < pos)*/
353  lmn_fatal("unexpected");
354  return false;
355  }
356  }
357 
358  bool write_direct(BYTE b, int pos) {
359  resize(pos);
360 
361  BS_SET(v, pos, b);
362  cur = pos + 1;
363  return true;
364  }

366  LmnBinStr *to_lmn_binstr() const {
367  struct LmnBinStr *ret_bs;
368  int size = (this->cur + 1) / 2;
369  ret_bs = LMN_MALLOC(struct LmnBinStr);
370  ret_bs->v = LMN_NALLOC(BYTE, size);
371  ret_bs->type = 0x00U;
372  memcpy(ret_bs->v, this->v, size);
373  ret_bs->len = this->cur;
374  if (ret_bs->len & 1) {
375  ret_bs->v[ret_bs->len >> 1] = ret_bs->v[ret_bs->len >> 1] & 0x0f;
376  }
377 
378  return ret_bs;
379  }

383  int org_size = size;
