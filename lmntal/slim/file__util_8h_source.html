73  path filename() const {
74  auto pos = str_.find_last_of(DIR_SEPARATOR_CHAR);
75  return (pos == std::string::npos)
76  ? *this
77  : str_.substr(pos + 1, str_.size() - pos - 1);
78  }
79  path stem() const { return str_.substr(0, str_.find_last_of('.')); }
80  path extension() const {
81  auto pos = str_.find_last_of('.');
82  return (pos == std::string::npos)
83  ? ""
84  : str_.substr(pos + 1, str_.size() - pos - 1);
85  }
86  path &replace_extension(const path &replacement = path()) {
87  str_ = str_.substr(0, str_.find_last_of('.')) + '.' + replacement.str_;
88  return *this;
89  }
90  std::string string() const { return str_; }
91 };

131 class directory_stream {
132  filesystem::path dirpath;
133  DIR *dir;
134 
135 public:
136  directory_stream(const std::string &name) : dirpath(name) {
137  dir = opendir(name.c_str());
138  if (!dir)
139  throw filesystem_error(std::error_code(errno, std::system_category()),
140  name);
141  }
142  ~directory_stream() throw() {
143  if (closedir(dir))
144  throw filesystem_error(std::error_code(errno, std::system_category()),
145  dirpath.string());
146  }
147 
148  long loc() const { return telldir(dir); }
149  directory_entry read() {
150  auto err = errno;
151  auto e = readdir(dir);
152  if (e == nullptr && err != errno)
153  throw filesystem_error(std::error_code(errno, std::system_category()),
154  dirpath.string());
155  return (e == nullptr) ? directory_entry() : directory_entry(*e, dirpath);
156  }
157  void seek(long loc) { seekdir(dir, loc); }
158 };
