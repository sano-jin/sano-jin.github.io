226  /* 状態srcと等価な状態を新たに構築して返す.
227  * srcに階層グラフ構造が割り当てられている場合, その階層グラフ構造までを,
228  * else: srcにバイナリストリングが割り当てられている場合,
229  * そのバイナリストリングを, 複製(即ち, deep copy)する. また,
230  * これに伴うencode関連の状態フラグもコピーするが,
231  * 状態空間構築のためのフラグはコピーしない.
232  * なお, 引数memがNULLではない場合は,
233  * これをsrcに対応する階層グラフ構造として扱う.　*/
234  State *duplicate(LmnMembraneRef mem) {
235  State *dst = new State();
236 
237  if (!is_binstr_user() && !mem) {
238  mem = state_mem();
239  }
240 
241  if (mem) {
242  dst->state_set_mem(mem->copy_ex());
243 #ifdef PROFILE
244  if (lmn_env.profile_level >= 3) {
245  profile_add_space(PROFILE_SPACE__STATE_MEMBRANE, mem->space());
246  }
247 #endif
248  } else if (state_binstr()) {
249  dst->state_set_binstr(lmn_binstr_copy(state_binstr()));
250  if (is_encoded()) {
251  dst->set_encoded();
252  }
253 #ifdef PROFILE
254  if (lmn_env.profile_level >= 3) {
255  profile_add_space(PROFILE_SPACE__STATE_OBJECT, sizeof(struct State));
256  }
257 #endif
258  }
259 
260  dst->hash = hash;
261 
262 #ifdef PROFILE
263  if (lmn_env.profile_level >= 3 && mem) {
264  profile_add_space(PROFILE_SPACE__STATE_MEMBRANE, mem->space());
265  profile_finish_timer(PROFILE_TIME__STATE_COPY);
266  }
267 #endif
268  return dst;

271  void succ_set(const std::vector<void *> &v) {
272  if (!v.empty() && !successors) {
273  unsigned int i;
274  successor_num = v.size();
275  successors = LMN_NALLOC(succ_data_t, successor_num);
276  for (i = 0; i < successor_num; i++) {
277  successors[i] = (succ_data_t)v.at(i);
278  }
279 #ifdef PROFILE
280  if (lmn_env.profile_level >= 3) {
281  profile_add_space(PROFILE_SPACE__TRANS_OBJECT,
282  sizeof(succ_data_t) * v.size());
283  profile_remove_space(PROFILE_SPACE__TRANS_OBJECT, 0);
284  }
285 #endif
286  }
287  }

289  void succ_add(succ_data_t succ) {
290  if (!successors) {
291  successors = LMN_NALLOC(succ_data_t, 1);
292  } else {
293  successors = LMN_REALLOC(succ_data_t, successors, successor_num + 1);
294  }
295  successors[successor_num] = succ;
296  successor_num++;
297  }

423  /* 状態sに対応する階層グラフ構造Xを,
424  * Xに対して一意なIDとなるバイナリストリングへエンコードする.
425  * エンコードしたバイナリストリングをsに割り当てる.
426  * sに対応する階層グラフ構造Xへのメモリ参照が消えるため,
427  * 呼び出し側でメモリ管理を行う.
428  * sが既にバイナリストリングを保持している場合は,
429  * そのバイナリストリングは破棄する. (ただし,
430  * sが既に一意なIDとなるバイナリストリングへエンコード済みの場合は何もしない.)
431  * 既に割当済みのバイナリストリングを破棄するため,
432  * sをハッシュ表に登録した後の操作はMT-unsafeとなる. 要注意. */
433  void calc_mem_encode() {
434  if (!this->is_encoded()) {
435  LmnBinStrRef mid;
436 
437  if (this->state_mem()) {
438  mid = lmn_mem_encode(this->state_mem());
439  } else if (this->state_binstr()) {
440  LmnMembraneRef m;
441 
442  m = lmn_binstr_decode(this->state_binstr());
443  mid = lmn_mem_encode(m);
444  this->free_binstr();
445  m->free_rec();
446  } else {
447  lmn_fatal("unexpected.");
448  }
449 
450  this->state_set_binstr(mid);
451  this->hash = binstr_hash(this->state_binstr());
452  this->set_encoded();
453  }
454  }

456  void calc_binstr_delta() {
457  LmnBinStrRef org = this->state_binstr();
458  if (org && this->state_D_ref()) {
459  LmnBinStrRef dif;
460  dif = state_binstr_D_compress(org, this->state_D_ref());
461  state_D_cache(this, org);
462  this->state_set_binstr(dif);
463  } else {
464  this->s_unset_d();
465  }
466  }

468  /* バイナリストリングorgと状態ref_sのバイナリストリングとの差分バイナリストリングを返す.
469  * orgのメモリ管理は呼出し側で行う. */
470  static inline LmnBinStrRef state_binstr_D_compress(LmnBinStrRef org,
471  State *ref_s) {
472  LmnBinStrRef ref, dif;
473 
474  ref = state_D_fetch(ref_s);
475  if (ref) {
476  dif = lmn_bscomp_d_encode(org, ref);
477  } else {
478  ref = ref_s->reconstruct_binstr();
479  dif = lmn_bscomp_d_encode(org, ref);
480  if (ref_s->s_is_d()) {
481  lmn_binstr_free(ref);
482  }
483  }
484 
485  return dif;
486  }

494  /* 状態sに対応するバイナリストリングを, sがrefする状態を基に再構築して返す. */
495  LmnBinStrRef reconstruct_binstr() {
496  if (!this->s_is_d())
497  return this->state_binstr();
498 
499  LmnBinStrRef ref = this->state_D_ref()->reconstruct_binstr();
500  LmnBinStrRef ret = lmn_bscomp_d_decode(ref, this->state_binstr());
501  LMN_ASSERT(this->state_D_ref());
502 
503  if (this->state_D_ref()->s_is_d())
504  lmn_binstr_free(ref);
505 
506  return ret;
507  }

509  /* 状態sに対応する階層グラフ構造memへのアドレスを返す.
510  * memがエンコードされている場合は, デコードしたオブジェクトのアドレスを返す.
511  * デコードが発生した場合のメモリ管理は呼び出し側で行う. */
512  LmnMembraneRef restore_membrane() {
513  return this->restore_membrane_inner(TRUE);
514  }
515 
516  /* delta-compression用のinner関数.
517  * flagが真の場合, デコード済みのバイナリストリングをキャッシュから取得する.
518  * キャッシュにバイナリストリングを置かないケースで使用する場合は,
519  * inner関数を直接呼び出し, flagに偽を渡しておけばよい. */
520  LmnMembraneRef restore_membrane_inner(BOOL flag) {
521  if (this->state_mem())
522  return this->state_mem();
523 
524  if (this->s_is_d()) {
525  LmnBinStrRef b =
526  (flag) ? state_D_fetch(this) : this->reconstruct_binstr();
527  return lmn_binstr_decode(b);
528  }
529 
530  LmnBinStrRef b = this->state_binstr();
531  if (lmn_env.tree_compress && !b) {
532  TreeNodeID ref;
533  tcd_get_root_ref(&this->tcd, &ref);
534  LMN_ASSERT(ref);
535  LMN_ASSERT(tcd_get_byte_length(&this->tcd) != 0);
536  b = lmn_bscomp_tree_decode((TreeNodeID)ref,
537  tcd_get_byte_length(&this->tcd));
538  }
539 
540  LMN_ASSERT(b);
541  return lmn_binstr_decode(b);
542  }
