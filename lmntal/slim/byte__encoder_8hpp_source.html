69  ByteEncoder(const Rule &rule)
70  : loc(0), cap(256), byte_seq(LMN_NALLOC(BYTE, cap)), rule(rule) {
71  /* load(rule.amatch); */

83  void load(const InstBlock &ib) {
84  if (ib.has_label())
85  label_loc[ib.label] = loc;

91  void resolve_labels() {
92  for (auto &p : loc_label_ref) {
93  auto loc = p.first;
94  auto label = p.second;
95  auto target_loc = label_loc.find(label);
96 
97  if (target_loc != label_loc.end()) {
98  write_at<LmnJumpOffset>(
99  target_loc->second - loc - sizeof(LmnJumpOffset), loc);
100  } else {
101  fprintf(stderr, "label not found L%d\n", label);
102  lmn_fatal("implementation error");
103  }
104  }
105  }

107  void expand_byte_sec() {
108  cap *= 2;
109  byte_seq = LMN_REALLOC(BYTE, byte_seq, cap);

120  void load(const Instruction &inst) {
121  write_forward<LmnInstrOp>(inst.id);
122  auto arg_num = inst.args.size();
123 
124  /* REMOVEATOMは引数の数が2と3の場合がある。第三引数の
125  ファンクタは無視する */
126  if (inst.id == INSTR_REMOVEATOM && arg_num == 3) {
127  arg_num = 2;
128  }
129 
130  for (int i = 0; i < arg_num; i++)
131  c17::visit(loader(*this), inst.args[i]);
132  }

134  /* 現在の位置に書き込TYPE型のデータを書き込む */
135  template <typename T> void write(T value) { write_at<T>(value, loc); }
136 
137  /* 現在の書き込み位置を移動する */
138  template <typename T> void move_by() { loc += sizeof(T); }
139 
140  /* write & move_by */
141  template <typename T> void write_forward(T value) {
142  write<T>(value);
143  move_by<T>();
144  }
145 
146  /* LCOの位置に書き込む */
147  template <typename T> void write_at(T value, size_t loc) {
148  while (loc + sizeof(T) >= cap)
149  expand_byte_sec();
150  *(T *)(byte_seq + loc) = (value);
151  }

209  void load(const il::instr_arg::inst_list &arg) {
210  /* 命令列の長さを求めるため、開始位置を記録する */
211  /* INSTR_NOTでサブ命令列の長さを知る必要がある */
212  auto start = loc;
213  move_by<LmnSubInstrSize>();
214 
215  for (auto &inst : arg.value)
216  load(inst);
217 
218  /* startの位置に現在の位置との差を書き込む */
219  auto t = loc;
220  loc = start;
221  write<LmnSubInstrSize>(t - (start + sizeof(LmnSubInstrSize)));
222  loc = t;
223  }
